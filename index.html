system_prompt = """

 In our system, a “reportback” is a user response confirming completion of an action and providing any requested details. Your task is to review each reportback according to the rubric below and then provide a single pass/fail rating.

Use “true” if the reportback is approved.

Use “false” if the reportback is declined.

Additionally, we want to flag any dangerous scenarios:

If the reportback references intent to self-harm, set references_self_harm = true.

If the reportback references intent to harm others, set references_harm_to_others = true.

Rubric for Evaluating Reportbacks
1 / F

Does not complete the requested action or statement

Clearly uses AI or images that are not their own

Demonstrates no effort, creativity, or enthusiasm

Shows no scale of impact
Rating: Reject ❌
Tags: AI or internet images = Irrelevant, Foul language or images = Inappropriate

2 / D

Partially completes the requested action/statement

Only addresses part of the prompt (missing key elements)

Repetitive or provides no additional insight

Matches someone else’s statement verbatim

Little to no scale of impact; minimal effort and creativity
Rating: Reject ❌
Tags: None

3 / C

Completes the basics of the requested action/statement

Minimal effort or creativity (e.g., only one sentence)

Enough to “pass,” but nothing more

Shows a small scale of impact
Rating: Approve ✅
Tags: None

4 / B

Fully completes both the action and the statement

Demonstrates creativity and effort

Clearly connects the statement to the action

May be overly long (200+ words) but shows substantial effort

Scale of impact is clear
Rating: Approve ✅
Tags: Good Submission

5 / A

Fully completes the action and statement

Clearly articulates personal and/or community impact

Shows depth, creativity, and genuine passion

Explains the significance and scale of impact in detail

Demonstrates clear care and effort
Rating: Approve ✅✅
Tags: Good for Brand %s

Final Output Requirements:

Provide true or false as your overall verdict.

If self-harm is referenced, set references_self_harm = true.

If harm to others is referenced, set references_harm_to_others = true.
"""

def get_openai_response(input):
    data = {
        'model': 'gpt-4o-mini',
        'messages': [
            {
                'role': 'system',
                'content': system_prompt
            },
            {
                'role': 'user',
                'content': input
            }
        ],
        'response_format': {
            'type': 'json_schema',
            'json_schema': {
                'name': 'intentions_moderation_response',
                'schema': {
                    'type': 'object',
                    'properties': {
                        'approve': {
                            'type': 'boolean'
                        },
                        'references_self_harm': {
                            'type': 'boolean'
                        },
                        'references_harm_to_others': {
                            'type': 'boolean'
                        }
                    },
                    'required': ['approve', 'references_self_harm', 'references_harm_to_others'],
                    'additionalProperties': False                
                },
                'strict': True
            }
        }
    }

    response = requests.post(
        url = url,
        headers = headers,
        json = data
    )

    return response.json()

# extract the label and value and concatenate them
def extract_and_concatenate(rb_data):
    # Check if the first item exists and has primary_text
    if rb_data and len(rb_data) > 0 and rb_data[0].get('primary_text'):
        primary_text = rb_data[0]['primary_text']
        label = primary_text.get('label', '')
        value = primary_text.get('value', '')
        # Concatenate label and value with a separator
        return f"{label}: {value}"
    return ""

# Apply the function and create new column
df['label_and_text'] = df['rb_parsed'].apply(extract_and_concatenate)
